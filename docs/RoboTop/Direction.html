def self.named(name)
  return const_get(name.to_s) if ALL.include?(name.to_s)

  raise InvalidDirectionError, "No such direction '#{name}'"
end

def ==(other)
  if other.respond_to?(:name)
    name == other.name
  else
    name.to_s.downcase == other.to_s.downcase
  end
end
